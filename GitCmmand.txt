cd -- переход к папке
ls или dir -- список файлов в папке
git clone <путь к репозиторию с сайта> (Вид: https://github.com/Cerega23rus/WA.git) --получить проект с сайта
git status -- статус 
git add <имя файла с расширением> -- подготовка добавления одного файла
git add -A -- подготовка добавления всех файлов
git commit -m "<описание изменений>" --добавление описания изменений
git push -- отправка на облако
git pull -- обновление всех файлов с облака
clear -- очистка строк в терминале
git -- список всех команд

esc потом wq и enter -- выход из окна с ошибкой
